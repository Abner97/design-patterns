Implementar los patrones de diseño

lib/patterns/PokeApiClient.ts

// CREACIONAL: Singleton
export class PokeApiClient {
  private static instance: PokeApiClient;
  private readonly baseUrl = "https://pokeapi.co/api/v2";

  private constructor() {}

  static getInstance(): PokeApiClient {
    if (!PokeApiClient.instance) {
      PokeApiClient.instance = new PokeApiClient();
    }
    return PokeApiClient.instance;
  }

  async get<T>(path: string): Promise<T> {
    const res = await fetch(`${this.baseUrl}${path}`, {
      next: { revalidate: 3600 },
    });
    if (!res.ok) throw new Error(`PokeAPI error: ${res.status}`);
    return res.json() as Promise<T>;
  }
}

lib/patterns/PokemonFacade.ts

// ESTRUCTURAL: Facade
import { PokeApiClient } from "./PokeApiClient";
import type { PokemonDetail, PokemonListResponse } from "../types";

export class PokemonFacade {
  private client = PokeApiClient.getInstance();

  async getList(limit = 20, offset = 0): Promise<PokemonListResponse> {
    return this.client.get(`/pokemon?limit=${limit}&offset=${offset}`);
  }

  async getDetail(nameOrId: string | number): Promise<PokemonDetail> {
    return this.client.get(`/pokemon/${nameOrId}`);
  }
}

lib/patterns/FavoritesStore.ts

// COMPORTAMIENTO: Observer
export interface FavoritesObserver {
  onFavoritesChange(favorites: string[]): void;
}

export class FavoritesStore {
  private static instance: FavoritesStore;
  private observers: FavoritesObserver[] = [];
  private favorites: Set<string>;

  private constructor() {
    const stored = typeof window !== "undefined"
      ? localStorage.getItem("poke-favorites") : null;
    this.favorites = new Set(stored ? JSON.parse(stored) : []);
  }

  static getInstance(): FavoritesStore {
    if (!FavoritesStore.instance)
      FavoritesStore.instance = new FavoritesStore();
    return FavoritesStore.instance;
  }

  subscribe(observer: FavoritesObserver): void {
    this.observers.push(observer);
  }

  unsubscribe(observer: FavoritesObserver): void {
    this.observers = this.observers.filter((o) => o !== observer);
  }

  toggle(name: string): void {
    this.favorites.has(name)
      ? this.favorites.delete(name)
      : this.favorites.add(name);
    const list = this.getAll();
    this.observers.forEach((o) => o.onFavoritesChange(list));
    localStorage.setItem("poke-favorites", JSON.stringify(list));
  }

  isFavorite(name: string): boolean { return this.favorites.has(name); }
  getAll(): string[] { return Array.from(this.favorites); }
}

Crear las API Routes de Next.js

Las API Routes actúan como proxy entre el cliente y la PokéAPI, heredando la caché del Singleton.

app/api/pokemon/route.ts

import { NextRequest, NextResponse } from "next/server";
import { PokemonFacade } from "@/lib/patterns/PokemonFacade";

const facade = new PokemonFacade();

export async function GET(req: NextRequest) {
  const { searchParams } = req.nextUrl;
  const limit = Number(searchParams.get("limit") ?? 20);
  const offset = Number(searchParams.get("offset") ?? 0);
  try {
    const data = await facade.getList(limit, offset);
    return NextResponse.json(data);
  } catch {
    return NextResponse.json(
      { error: "Failed to fetch pokemon list" },
      { status: 500 }
    );
  }
}

app/api/pokemon/[name]/route.ts

import { NextRequest, NextResponse } from "next/server";
import { PokemonFacade } from "@/lib/patterns/PokemonFacade";

const facade = new PokemonFacade();

export async function GET(
  _req: NextRequest,
  { params }: { params: Promise<{ name: string }> }
) {
  const { name } = await params;
  try {
    const data = await facade.getDetail(name);
    return NextResponse.json(data);
  } catch {
    return NextResponse.json(
      { error: "Pokemon not found" },
      { status: 404 }
    );
  }
}

Estrategia de renderizado

Cada ruta usa la estrategia correcta según sus necesidades.

Ruta Estrategia Motivo
/ Client Component Paginación con estado del cliente
/pokemon/[name] SSR SEO + metadata dinámica
/favorites Client Component Lee localStorage vía Observer
/api/pokemon API Route Proxy + caché para la lista
/api/pokemon/[name] API Route Proxy + caché para detalles

Configurar y ejecutar

Habilita las imágenes de la PokéAPI en next.config.ts:

import type { NextConfig } from "next";

const nextConfig: NextConfig = {
  images: {
    remotePatterns: [{
      protocol: "https",
      hostname: "raw.githubusercontent.com",
      pathname: "/PokeAPI/**",
    }],
  },
};

export default nextConfig;

Luego ejecuta:

npm run dev
